Modular Dashboard Platform (MDP)
Technical Design Document v1.0

1. Executive Summary
The Modular Dashboard Platform (MDP) is a scalable, containerized web application that enables users to create dynamic, personalized dashboards with modular windows (widgets) that can connect to various data sources (databases, APIs). The system features real-time asynchronous operations, multi-tenant user management with role-based access control (RBAC), theme switching, layout persistence via JSON definitions, comprehensive logging, and versioning capabilities.
Key Architectural Decisions: - Frontend: React 18+ with TypeScript, React-Grid-Layout for drag-drop-resize functionality - Backend: Node.js (Express/Fastify) or Python (FastAPI) with async/await patterns throughout - Database: PostgreSQL (primary) + Redis (caching/sessions) + MongoDB (layout storage) - Containerization: Docker with multi-stage builds for optimized images - Orchestration: Docker Compose (dev) / Kubernetes (production)

// RBAC Middlewareexport const requirePermission = (permission: Permission) => {  return async (req: Request, res: Response, next: NextFunction) => {    const userId = req.user.id;        // Check cache first    const cacheKey = `permissions:${userId}`;    let permissions = await redis.get(cacheKey);        if (!permissions) {      // Fetch from DB      const { rows } = await pgPool.query(        `SELECT DISTINCT jsonb_array_elements_text(r.permissions) as perm         FROM users u         JOIN user_roles ur ON u.id = ur.user_id         JOIN roles r ON ur.role_id = r.id         WHERE u.id = $1 AND u.is_active = true`,        [userId]      );            permissions = rows.map(r => r.perm);      await redis.setex(cacheKey, 300, JSON.stringify(permissions));    } else {      permissions = JSON.parse(permissions);    }        if (!permissions.includes(permission) && !permissions.includes('*')) {      return res.status(403).json({ error: 'Insufficient permissions' });    }        next();  };};// Usage in routesrouter.post('/layouts',   authenticate,   requirePermission(Permission.LAYOUT_CREATE),  asyncHandler(layoutController.createLayout));

async function createLayoutVersion(layoutId: string, userId: string, summary: string) {  const layout = await getLayout(layoutId);  const versionNumber = layout.version + 1;    // Store snapshot in PostgreSQL  await pgPool.query(    `INSERT INTO layout_versions (layout_id, version_number, created_by,      change_summary, layout_snapshot)     VALUES ($1, $2, $3, $4, $5)`,    [layoutId, versionNumber, userId, summary, JSON.stringify(layout)]  );    // Update current version in MongoDB  await mongoClient.db().collection('layouts').updateOne(    { _id: new ObjectId(layoutId) },    { $set: { version: versionNumber, updated_at: new Date() } }  );    // Keep only last 50 versions (cleanup in background)  cleanupOldVersions(layoutId);    return versionNumber;}
